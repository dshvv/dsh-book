👉redux-toolkit 官方文档 👈

这是啥​

redux-toolkit 是 Redux 官方出的工具包，用于高效的 Redux 开发。

yarn add @reduxjs/toolkit

chrome 插件支持​

原生的写法不支持浏览器使用插件进行跟踪 debug。redux-toolkit 创建的 store 则可以

/src/store/index.jsx
import { configureStore } from "@reduxjs/toolkit";

// state
const initialState = {
  msg: "张三",
};

// reducer
const reducer = (state = initialState, action) => {
  switch (action.type) {
    case "SET_MSG":
      return { ...state, ...{ msg: action.payload } };
    default:
      return state;
  }
};

// store
export default configureStore({ reducer });

编写 action creator​

src/store/action-creator.js
import { createAction as createActionCreator } from "@reduxjs/toolkit";

 /**
  * 利用createActionCreator来定义一个actioCreator，用于 创建type为SET_MSG 的action,
  * @param payload为action中的状态属性
  * @return action
  */
export const setMsg = createActionCreator('SET_MSG');

编写 reducer​

src/store/index.js
import { configureStore, createReducer } from "@reduxjs/toolkit";
import { setMsg } from './action-creator'

// state
const initialState = {
  msg: '张三'
};

// reducer
const reducer = createReducer(initialState, (builder) => {
  builder
    .addCase(setMsg, (state, action) => {
      state.msg = action.payload;
    })
})

// store
export default configureStore({ reducer });

使用切片​

当然还有更进一步的优化写法，就是切片 Slice。
工具包提供一个 createSlice 函数，它接受初始状态、reducer 函数的对象，并自动生成 reducer 和 action creators 和 action types。
如此一来便不再需要辅助函数creatActionCreator、creatReducer，以及定义action-creator了。 来让我们使用切片来定义一个 store

/src/store/index.jsx
import { configureStore, createSlice } from "@reduxjs/toolkit";

// state
const initialState = {
  msg: "张三",
};

// slice
// slice包含 action、reducer,如果有需要可以后期结构出来用
const slice = createSlice({
  name: "app",
  initialState,
  reducers: {
    setMsg(state, action) {
      state.msg = action.payload;
    },
  },
});

// actionCreator
// 工具会根据切片里的reducers自动生成与其同名的actionCreator
// actionCreator通过（同样自动生成的）type和reducer进行关联，来达到更新state的目的
export const { setMsg } = slice.actions;

// store
export default configureStore({
  reducer: slice.reducer,
});
